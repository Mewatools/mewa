107 inline bool MxMatrix::isIdentity() const
108 {
109  if (flagBits == Identity)
110  return true;
111  if (m[0][0] != 1.0f || m[0][1] != 0.0f || m[0][2] != 0.0f)
112  return false;
113  if (m[0][3] != 0.0f || m[1][0] != 0.0f || m[1][1] != 1.0f)
114  return false;
115  if (m[1][2] != 0.0f || m[1][3] != 0.0f || m[2][0] != 0.0f)
116  return false;
117  if (m[2][1] != 0.0f || m[2][2] != 1.0f || m[2][3] != 0.0f)
118  return false;
119  if (m[3][0] != 0.0f || m[3][1] != 0.0f || m[3][2] != 0.0f)
120  return false;
121  return (m[3][3] == 1.0f);
122 }

429 inline MxVector3F operator*(const MxMatrix& matrix, const MxVector3F& vector)
430 {
431  float x, y, z, w;
432  if (matrix.flagBits == MxMatrix::Identity) {
433  return vector;
434  } else if (matrix.flagBits < MxMatrix::Rotation2D) {
435  // Translation | Scale
436  return MxVector3F(vector.x() * matrix.m[0][0] + matrix.m[3][0],
437  vector.y() * matrix.m[1][1] + matrix.m[3][1],
438  vector.z() * matrix.m[2][2] + matrix.m[3][2]);
439  } else if (matrix.flagBits < MxMatrix::Rotation) {
440  // Translation | Scale | Rotation2D
441  return MxVector3F(vector.x() * matrix.m[0][0] + vector.y() * matrix.m[1][0] + matrix.m[3][0],
442  vector.x() * matrix.m[0][1] + vector.y() * matrix.m[1][1] + matrix.m[3][1],
443  vector.z() * matrix.m[2][2] + matrix.m[3][2]);
444  } else {
445  x = vector.x() * matrix.m[0][0] +
446  vector.y() * matrix.m[1][0] +
447  vector.z() * matrix.m[2][0] +
448  matrix.m[3][0];
449  y = vector.x() * matrix.m[0][1] +
450  vector.y() * matrix.m[1][1] +
451  vector.z() * matrix.m[2][1] +
452  matrix.m[3][1];
453  z = vector.x() * matrix.m[0][2] +
454  vector.y() * matrix.m[1][2] +
455  vector.z() * matrix.m[2][2] +
456  matrix.m[3][2];
457  w = vector.x() * matrix.m[0][3] +
458  vector.y() * matrix.m[1][3] +
459  vector.z() * matrix.m[2][3] +
460  matrix.m[3][3];
461  if (w == 1.0f)
