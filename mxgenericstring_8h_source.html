393  void appendNumber( int number )
394  {
395  Q_ASSERT( NULL != d );
396  Q_ASSERT( d->refCounter >= 0 );
397 
398  DynamicData *dd = static_cast<DynamicData*>(d);
399  T * dstPtr = data();
400  dstPtr += dd->size;
401  if( number >= 0 && number <= 99 )
402  {
403  Q_ASSERT( dd->alloc >= (dd->size + 1) );
404 
405  int firstDigit = number / 10;
406  int secondDigit = number % 10;
407  T digitChar;
408  if( firstDigit > 0 )
409  {
410  digitChar = firstDigit + 48;
411  *dstPtr = digitChar;
412  dd->size++;
413  dstPtr++;
414  }
415 
416  Q_ASSERT( dd->alloc >= (dd->size + 1) );
417 
418 
419  digitChar = secondDigit + 48;
420  *dstPtr = digitChar;
421  dd->size++;
422  }
423  else
424  {
425  // check how long will be the string
426  bool changedSign = false;
427  int stringSize = 0;
428  if( number < 0 ) {
429  changedSign = true;
430  stringSize++;
431  number *= -1;
432  }
433 
434  int shifter = number;
435 
436  do{ //Move to where representation ends
437  ++stringSize;
438  shifter = shifter/10;
439  }while(shifter);
440 
441  if( (dd->size + stringSize) >= dd->alloc )
442  {
443  int newSize = dd->size + stringSize + 10;
444  reserveTotal(newSize);
445  }
446 
447 
448  dd->size += stringSize;
449 
450  char const digit[] = "0123456789";
451  if(changedSign) {
452  *dstPtr = '-';
453  dstPtr++;
454  stringSize--;
455  }
456 
457  dstPtr += (stringSize-1);
458  do{ //Move back, inserting digits as u go
459  *dstPtr = digit[number%10];
460  dstPtr--;
461  number = number/10;
462  stringSize--;
463  }while(stringSize);
464 
465 
466 
467  }
468  }
