80  void rotateBy( T radians ) // rotate relative to 0,0
81  {
82  T s = sin(radians);
83  T c = cos(radians);
84 
85  // rotate point
86  T tmp = pVector[0] * c - pVector[1] * s;
87  pVector[1] = pVector[0] * s + pVector[1] * c;
88  pVector[0] = tmp;
89  }
