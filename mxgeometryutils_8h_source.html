59 inline void lineIntersection( const MxVector2F &ps1, const MxVector2F &pe1, const MxVector2F &ps2, const MxVector2F &pe2, MxVector2F &result )
60 {
61  // Get A,B,C of first line - points : ps1 to pe1
62  float A1 = pe1.y() -ps1.y();
63  float B1 = ps1.x() - pe1.x();
64  float C1 = A1*ps1.x() + B1*ps1.y();
65 
66  // Get A,B,C of second line - points : ps2 to pe2
67  float A2 = pe2.y() - ps2.y();
68  float B2 = ps2.x() - pe2.x();
69  float C2 = A2*ps2.x() + B2*ps2.y();
70 
71  // Get delta and check if the lines are parallel
72  float delta = A1*B2 - A2*B1;
73  Q_ASSERT(delta != 0.0f); // Lines are parallel
74 
75  // now return the Vector2 intersection point
76  result = MxVector2F( (B2*C1 - B1*C2)/delta, (A1*C2 - A2*C1)/delta );
77 }

106 inline void rotatePoint( const MxVector2F &center, MxVector2F &p, float angle )
107 {
108  float s = sin(angle);
109  float c = cos(angle);
110 
111 
112  // translate point back to origin:
113  p[0] -= center.x();
114  p[1] -= center.y();
115 
116  // rotate point

126 inline float solveLineY(const MxVector2F &p1, const MxVector2F &p2, float x )
127 {
128  Q_ASSERT( (p1.x() <= x && x <= p2.x()) || (p2.x() <= x && x <= p1.x()) );
129  /* This is an optimization to calculate the line equation.
130  It translates one point of the line to the origin
131  so it avoids calculating b. */
132  const MxVector2F delta = p2 - p1;
133  float m = delta.y() / delta.x();
134  float yValue = m*(x - p1.x());
135  yValue += p1.y();
136  return yValue;
137 }

142 {
143  float imageWidth = from.width();
144  float imageHeight = from.height();
145  float viewWidth = to.width();
146  float viewHeight = to.height();
147  float x, y, w, h;
148 
149  float scaledWidth = (imageWidth / imageHeight) * viewHeight;
150  bool useHeight = (scaledWidth <= viewWidth);
151 
152  if (useHeight)
153  {
154  w = scaledWidth;
155  h = viewHeight;
156  x = (viewWidth - w) / 2.0 ;
157  y = 0.0f;
158  }
159  else
160  {
161  h = imageHeight / imageWidth * viewWidth;
162  w = viewWidth;
163  x = 0.0f;
164  y = (viewHeight - h) / 2.0f;
165  }
166 
167  return MxRect<T>(x, x+w, y, y+h);
168 }
