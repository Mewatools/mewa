457  //Q_ASSERT(d->ref == 1);

470 // ensures that enough space is available to append n elements
471 template <typename T, typename Initializer, int Prealloc>
472 T* MxList<T,Initializer,Prealloc>::p_append(int n)
473 {
474  Q_ASSERT(d->refCounter == 1);
475  int e = d->end;
476  if (e + n > d->alloc) {
477  int b = d->begin;
478  if (b - n >= 2 * d->alloc / 3) {
479  // we have enough space. Just not at the end -> move it.
480  e -= b;
481  memcpy(d->array, d->array + b, e * sizeof(T));
482  d->begin = 0;
483  } else {
484  grow(d->alloc + n);
485  }
486  }
487  d->end = e + n;
488  return d->array + e;
489 }

492 template <typename T, typename Initializer, int Prealloc>
493 T* MxList<T,Initializer,Prealloc>::p_prepend()
494 {
495  Q_ASSERT(d->refCounter == 1);
496  if (d->begin == 0) {
497  if (d->end >= d->alloc / 3)
498  grow(d->alloc + 1);
499 
500  if (d->end < d->alloc / 3)
501  d->begin = d->alloc - 2 * d->end;
502  else
503  d->begin = d->alloc - d->end;
504 
505  memmove(d->array + d->begin, d->array, d->end * sizeof(T));
506  d->end += d->begin;
507  }
508  return d->array + --d->begin;
509 }

511 template <typename T, typename Initializer, int Prealloc>
512 inline T* MxList<T,Initializer,Prealloc>::p_insert(int i)
513 {
514  //Q_ASSERT(d->ref == 1);
515  if (i <= 0)
516  return p_prepend();
517  int size = d->end - d->begin;
518  if (i >= size)
519  return p_append(1);
520 
521  bool leftward = false;
522 
523  if (d->begin == 0) {
524  if (d->end == d->alloc) {
525  // If the array is full, we expand it and move some items rightward
526  grow(d->alloc + 1);
527  } else {
528  // If there is free space at the end of the array, we move some items rightward
529  }
530  } else {
531  if (d->end == d->alloc) {
532  // If there is free space at the beginning of the array, we move some items leftward
533  leftward = true;
534  } else {
535  // If there is free space at both ends, we move as few items as possible
536  leftward = (i < size - i);
537  }
538  }
539 
540  if (leftward) {
541  --d->begin;
542  memmove(d->array + d->begin, d->array + d->begin + 1, i * sizeof(T));
543  } else {
544  memmove(d->array + d->begin + i + 1, d->array + d->begin + i,
545  (size - i) * sizeof(T));
546  ++d->end;
547  }
548  return d->array + d->begin + i;
549 }

627 template <typename T, typename Initializer, int Prealloc>
628 inline typename MxList<T,Initializer,Prealloc>::Data* MxList<T,Initializer,Prealloc>::p_detach_grow(int *idx, int num)
629 {
630  Data *x = d;
631  int l = x->end - x->begin;
632  int nl = l + num;
633  int alloc = grow(nl);
634  Data* t = static_cast<Data *>(malloc(DataHeaderSize + alloc * sizeof(T)));
635 
636  t->refCounter = 1;
637  t->alloc = alloc;
638  // The space reservation algorithm's optimization is biased towards appending:
639  // Something which looks like an append will put the data at the beginning,
640  // while something which looks like a prepend will put it in the middle
641  // instead of at the end. That's based on the assumption that prepending
642  // is uncommon and even an initial prepend will eventually be followed by
643  // at least some appends.
644  int bg;
645  if (*idx < 0) {
646  *idx = 0;
647  bg = (alloc - nl) >> 1;
648  } else if (*idx > l) {
649  *idx = l;
650  bg = 0;
651  } else if (*idx < (l >> 1)) {
652  bg = (alloc - nl) >> 1;
653  } else {
654  bg = 0;
655  }
656  t->begin = bg;
657  t->end = bg + nl;
658  d = t;
659 
660  return x;
661 }
